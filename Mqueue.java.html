import java.util.*;
import java.util.Queue;

public class Mqueue {
    public static void main(String[] args) {
        //Array for the queue
        Integer[] numArray = {3, 17, 34, 5, 19};
        //initialize the queue
        Queue<Integer> queue = new LinkedList<>();
        //for loop to put the array numbers into the queue one by one
        for (Integer element : numArray) ;
        Collections.addAll(queue, numArray);
        //Prints the queue in full
        System.out.println("queue after adding elements " + queue);
        //Removes the first element in the queue
        try {
            Integer removedElement = queue.remove();
            //Prints the removed element and the updated queue without the element
            System.out.println("Element removed: " +removedElement);
            System.out.println("Updated queue" +queue);
            //Error if the queue is empty
        }catch (NoSuchElementException e){
            System.out.println("Error: queue is empty");
        }
        //Shows the first element in the queue without removing it
        try{
            Integer frontElement = queue.peek();
            //Prints the first element in the que
            System.out.println("Current front element: " +frontElement);
            //Error if the queue is empty
        }catch (NoSuchElementException e){
            System.out.println("Error: queue is empty");
        }
        //Checks and prints the queue size
        int size = queue.size();
        System.out.println("Queue size:" +size);
        //Checks if the queue is empty or not and returns true or false
        Boolean isEmptyQueue = queue.isEmpty();
        System.out.println("Is the queue empty? " +isEmptyQueue);
        }


    }
